function y=get_final_loc(N)
    x0=0; y0=0;
    x = zeros(2,N);

    for i=1:1:N
        l = get_stride(1);
        th = get_angle(1);
        deltax = l*cos(th); deltay = l*sin(th);
        x(1,i) = deltax;
        x(2,i) = deltay;
    end
    xinc = x(1,:); yinc = x(2,:);
    xtraj = sum(xinc); ytraj = sum(yinc); rorigin = (xtraj.^2 + ytraj.^2).^0.5;
    y= rorigin;
end
